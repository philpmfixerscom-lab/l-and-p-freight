""" ELD / Driver Mobile View — minimal self-contained web app. Shows: - Current dispatch / BOL - In-cab bill info - HOS clock - Stub location Run: python -m http.server 8080 from ./eld_mobile """ from __future__ import annotations import json import math import random from datetime import datetime from pathlib import Path BASE_DIR = Path(__file__).resolve().parent STUB_DATA_PATH = BASE_DIR / "stub_data.json" def _now_iso() -> str: return datetime.now().isoformat(timespec="seconds") def _seed_stub_data() -> dict[str, Any]: if STUB_DATA_PATH.exists(): return json.loads(STUB_DATA_PATH.read_text()) data = { "driver": "Phillip / Lawson", "vehicle": "39ft End-Dump + Tractor", "current_load": { "bol_number": "LP-20260706001", "shipper": "Sibelco", "commodity": "Feldspar", "weight_tons": 24, "origin": "Spruce Pine, NC", "destination": "Central Georgia (Kohler)", "status": "In Transit", "rate_per_ton": 48.0, "total_revenue": 1152.0, }, "location": { "lat": 35.912, "lon": -82.064, "speed_mph": 58.0, "heading_deg": 185.0, }, "hos": { "hours_today": 3.5, "hours_week": 22.0, "drive_remaining_hours": 7.5, "on_duty_remaining_hours": 10.5, "cycle_remaining_hours": 38.0, }, } STUB_DATA_PATH.write_text(json.dumps(data, indent=2)) return data def _haversine_miles(lat1: float, lon1: float, lat2: float, lon2: float) -> float: r = 3958.8 phi1 = math.radians(lat1) phi2 = math.radians(lat2) dphi = math.radians(lat2 - lat1) dlambda = math.radians(lon2 - lon1) a = math.sin(dphi / 2) ** 2 + math.cos(phi1) * math.cos(phi2) * math.sin(dlambda / 2) ** 2 return 2 * r * math.asin(math.sqrt(min(1.0, a))) DEST_LAT = 32.98 DEST_LON = -82.72 def render_driver_mobile() -> str: data = _seed_stub_data() loc = data["location"] miles_to_dest = _haversine_miles(loc["lat"], loc["lon"], DEST_LAT, DEST_LON) eta_hours = miles_to_dest / max(loc["speed_mph"], 1.0) eta = datetime.now().strftime("%I:%M %p") if eta_hours < 0.1 else ( datetime.now().timestamp() + eta_hours * 3600 ) eta_str = datetime.fromtimestamp(eta).strftime("%I:%M %p") if isinstance(eta, float) else eta hos = data["hos"] hos_items = [ ("Drive remaining", f"{hos['drive_remaining_hours']}h"), ("On-duty remaining", f"{hos['on_duty_remaining_hours']}h"), ("Cycle remaining", f"{hos['cycle_remaining_hours']}h"), ("Hours today", f"{hos['hours_today']}h"), ("Hours week", f"{hos['hours_week']}h"), ] load = data["current_load"] return f""" L & P Driver
